    corpus: Corpus,
    *,
    skip_download: bool = False,
    workers: int = DEFAULT_WORKERS,

) -> None:
    """Fetch docs using the corpus's configured fetcher plugin."""
    if skip_download:

        return

**3. Rewrite `run_embed()`:**

```python
async def run_embed(
    corpus: Corpus,
    chunks: list | None = None,
    *,
    embedder=None,
) -> list:
    """Embed chunks using the corpus's configured embedder plugin."""
    import json
    from doc_hub.embed import embed_chunks
    from doc_hub.parse import Chunk

    log.info("[%s] === STEP 3: Embed (embedder=%s) ===", corpus.slug, corpus.embedder)

    # Load chunks from JSONL if not provided
    if chunks is None:
        chunks_path = chunks_dir(corpus) / "chunks.jsonl"
        if not chunks_path.exists():
            raise FileNotFoundError(
                f"chunks.jsonl not found at {chunks_path}. Run parse first."
            )
        loaded: list[Chunk] = []
        with chunks_path.open(encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if line:
                    loaded.append(Chunk(**json.loads(line)))
        chunks = loaded
        log.info("[%s] Loaded %d chunks from %s", corpus.slug, len(chunks), chunks_path)

    # Resolve embedder plugin
    if embedder is None:
        from doc_hub.discovery import get_registry
        embedder = get_registry().get_embedder(corpus.embedder)

    embedded = await embed_chunks(corpus.slug, chunks, embedder)
    log.info("[%s] Embed complete → %d embedded chunks", corpus.slug, len(embedded))
    return embedded
```

async def _add_corpus_impl(
    *,
    slug: str,
    name: str,
    strategy: str,
    config: dict,
    parser: str = "markdown",
    embedder: str = "gemini",
    pool: asyncpg.Pool,
) -> dict:
    """Register a new corpus with plugin names for each stage."""
    # Optionally validate plugin names exist in registry
    from doc_hub.discovery import get_registry
    registry = get_registry()

    # Warn (don't error) if plugins aren't found — they might be
    # installed later before the pipeline runs
    for name_val, kind in [(strategy, "fetcher"), (parser, "parser"), (embedder, "embedder")]:
        available = getattr(registry, f"list_{kind}s")()
        if name_val not in available:
            log.warning(
                "Plugin %s %r not currently registered. "
                "Available: %s. Install the plugin before running the pipeline.",
                kind, name_val, available,
            )

    corpus = Corpus(
        slug=slug,
        name=name,
        fetch_strategy=strategy,
        fetch_config=config,
        parser=parser,
        embedder=embedder,
    )
    await db.upsert_corpus(pool, corpus)

## Design Decisions

1. **Per-corpus embedder resolution**: Each corpus names its embedder in the `embedder` field. The pipeline resolves it from the registry. This means different corpora can use different embedders (as long as all embedders produce the same vector dimension — enforced by `get_vector_dim()`).

2. **No `gemini_client` passthrough**: The old pattern of passing `gemini_client` through the call chain is gone. Plugins manage their own clients. The `GeminiEmbedder` creates its client lazily.

3. **Soft validation in `add_corpus`**: When registering a corpus, missing plugins produce a warning, not an error. This allows registering a corpus before its plugin package is installed (e.g., in a deployment script that installs plugins in a separate step).
